% MOTIVATION
% ============================================================================
\begin{frame}{Motivation: Functional Safety Implies Bounded Time}
\begin{itemize}
\item Functional safety requires detecting and controlling faults before hazards occur.
\item The fault-tolerant time interval (\FTTI) bounds how long we can take to \emph{detect + react}.
\item Modern CPS/automotive SoCs already include a \SafetyIsland: isolated compute that can supervise the main system and force a fail-safe reaction.
\end{itemize}

\vspace{0.6em}
\begin{block}{Opportunity}
Move beyond watchdogs and fault flags to \hl{online policy monitoring} over real signals.
\end{block}
\end{frame}

% ============================================================================
% SAFETY ISLANDS
% ============================================================================
\begin{frame}{Safety Islands: Where They Sit and What They Do}
\begin{itemize}
\item \textbf{Isolation.} Separate core(s), memory region, and trusted software stack.
\item \textbf{Supervision.} Observe system health and safety-relevant signals; decide when to trigger a safe state.
\item \textbf{Interfaces.} Today, many interfaces are coarse (fault flags, watchdog kicks, periodic status).
\end{itemize}

\vspace{0.6em}
\begin{alertblock}{Missing Today}
Architecture support to let the \SafetyIsland check \emph{behavioral safety policies} fast enough to matter.
\end{alertblock}
\end{frame}

% ============================================================================
% POLICY MONITORING
% ============================================================================
\begin{frame}{Policy Monitoring: Intent vs.\ Evidence}
\begin{columns}[T,onlytextwidth]
  \column{0.50\textwidth}
  \begin{block}{Intent}
  What the controller asked for:
  \begin{itemize}
  \item Duty cycle, dead-time
  \item Update sequence, timestamp
  \end{itemize}
  \end{block}

  \column{0.50\textwidth}
  \begin{alertblock}{Evidence}
  What the system actually did:
  \begin{itemize}
  \item Measured PWM edges
  \item Waveform timing
  \end{itemize}
  \end{alertblock}
\end{columns}

\vspace{0.6em}
\begin{itemize}
\item Policies are temporal constraints over signals; \STL is one common encoding.
\item The \SafetyIsland correlates them and triggers a reaction if the policy is violated.
\end{itemize}
\end{frame}

% ============================================================================
% STATE OF PRACTICE
% ============================================================================
\begin{frame}{State of Practice: DMA/IPC Safety-Island Interfaces}
\begin{itemize}
\item Most current \SafetyIsland designs ingest data via \hl{IPC mailboxes} and DMA-friendly shared buffers.
\item Examples: Arm Kronos documents HIPC cross-domain messaging; NVIDIA DriveOS describes a functional safety island with controlled channels.
\item For step-by-step intent export, these paths often require extra copies or cache maintenance.
\item That overhead and jitter competes directly with tight \FTTI budgets.
\end{itemize}

\vspace{0.4em}
\begin{block}{Our Proposal}
Use a \hl{small cache-coherent shared region} as the safety-island interface for runtime policy monitoring.
\end{block}
\end{frame}

% ============================================================================
% PROBLEM STATEMENT
% ============================================================================
\begin{frame}{Problem Statement}
Check safety policies within \FTTI, using intent and evidence that live in different protection domains.

\vspace{0.6em}
\textbf{Constraints.}
\begin{itemize}
\item \keyword{Low latency:} the monitor must see new controller intent quickly.
\item \keyword{Low overhead:} intent export should not burden the control loop.
\item \keyword{Correctness:} the island must read a \emph{consistent} controller record (not a mixed snapshot).
\item \keyword{Predictability:} worst-case export + monitoring time must be bounded under contention.
\end{itemize}
\end{frame}

% ============================================================================
% GOAL AND HYPOTHESIS
% ============================================================================
\begin{frame}{Goal and Hypothesis}
\begin{alertblock}{Hypothesis}
\hl{Cache-coherent shared memory} is the right substrate for low-latency intent export from the \PrimaryCompute to an MCU-like \SafetyIsland monitor.
\end{alertblock}

\vspace{0.6em}
\textbf{Project goal.} Define the \hl{architecture contract} that makes coherent monitoring both correct and time-bounded:
\begin{itemize}
\item coherent interface choice for an MCU-like \SafetyIsland,
\item synchronization discipline for multi-field records,
\item predictability mechanisms to meet \FTTI.
\end{itemize}
\end{frame}

% ============================================================================
% WHY CACHE-COHERENT
% ============================================================================
\begin{frame}{Why Cache-Coherent Shared Memory?}
\textbf{Goal:} export controller intent to the \SafetyIsland with \hl{minimal latency and overhead}.

\vspace{0.4em}
\begin{itemize}
\item \textbf{Single-copy path:} controller writes once; island reads via ordinary loads (no message copying).
\item \textbf{Compared to IPC/DMA:} avoids extra copies and software-managed cache flush/invalidate on the fast path.
\item \textbf{Good fit for control:} frequent small records, tight deadlines.
\end{itemize}

\vspace{0.5em}
\begin{block}{Key Insight}
Cache coherence lets both cores observe a single up-to-date value \emph{per memory location} without explicit cache maintenance. This proposal adds \hl{snapshot correctness} and \hl{bounded latency} for the coherent export path.
\end{block}
\end{frame}

% ============================================================================
% RELATED WORK
% ============================================================================
\begin{frame}[shrink=5]{Related Work and Gap}
\begin{itemize}
\item \keyword{Safety islands (practice):} Arm Kronos/HIPC and NVIDIA DriveOS describe isolated islands, typically using IPC/DMA-style channels.
\item \keyword{Coherent fabrics (standards):} Arm AMBA ACE/CHI provide coherence-capable interconnects and coherent I/O options.
\item \keyword{Correctness under weak ordering:} Arm memory-model work (e.g., Pulte et al.) and heterogeneous consistency work (e.g., MemGlue, Compound Consistency).
\item \keyword{Bounded latency under contention:} predictable coherence (e.g., Hassan et al., Kaushik et al.) and mixed-criticality sharing vs.\ isolation (e.g., Chisholm et al.).
\item \keyword{Monitoring foundations:} Simplex-style runtime assurance and temporal monitors (e.g., RTAMT), but they often assume the trace is already well-timed.
\end{itemize}

\begin{alertblock}{Gap}
An end-to-end architecture for \hl{coherent intent export} + time-aligned evidence + bounded detection latency for safety-island policy monitoring.
\end{alertblock}
\end{frame}

% CHALLENGE 1: CONSISTENT RECORDS
% ============================================================================
\begin{frame}{Challenge 1: Consistent Records}
\textbf{Weak ordering:} other cores may observe your writes in a different order than your program order (unless you synchronize).

\vspace{0.6em}
\begin{columns}[T,onlytextwidth]
  \column{0.50\textwidth}
  \begin{block}{Writer (controller)}
  {\ttfamily
  store duty := new\quad\checkmark\\
  store ts\;\;\; := new\quad\checkmark}
  \end{block}

  \column{0.50\textwidth}
  \begin{alertblock}{Reader (\SafetyIsland)}
  {\ttfamily
  load ts\;\;\; == new\quad\checkmark\\
  load duty == old\quad$\times$}\\[2pt]
  {\small\color{Accent} mixed snapshot!}
  \end{alertblock}
\end{columns}

\vspace{0.8em}
We need an \hl{export record + synchronization discipline} so island reads behave like consistent ``steps.''
\end{frame}

\vspace{-0.3em}
\begin{itemize}
% \item \keyword{Export:} coherence traffic (invalidations, line transfers, directory lookups) + interconnect arbitration.
%       \muted{Dominated by cross-cluster snoop latency under contention.}
\item \keyword{Export:} coherence traffic
      \muted{Time it takes to propagate coherent data.}
\item \keyword{Monitor:} policy evaluation over the current signal window.
      \muted{Bounded by the policy horizon and evaluation complexity.}
\item \keyword{React:} triggering the fail-safe output.
      \muted{Hardware path; typically small and deterministic.}
\end{itemize}

\vspace{-0.2em}
Contention inflates $\Delta_{\text{export}}$; we need \hl{predictability mechanisms} (partitioning, priorities, protocol choices) to bound the worst case.
\end{frame}

% RESEARCH QUESTIONS
% ============================================================================
\begin{frame}{Research Questions}
\begin{enumerate}
\item[\keyword{RQ1}] \textbf{Coherent interface.} How should an MCU-like \SafetyIsland participate in coherence with an A-profile \PrimaryCompute?
\item[\keyword{RQ2}] \textbf{Snapshot correctness.} What record format + synchronization rules guarantee consistent multi-field snapshots under weak ordering?
\item[\keyword{RQ3}] \textbf{Predictability.} What worst-case export and monitoring latency bounds are achievable under contention, and what mechanisms enforce them?
\item[\keyword{RQ4}] \textbf{Evidence and time alignment.} Where do we measure realized actuation, and how do we align evidence with controller records?
\end{enumerate}
\end{frame}

% CASE STUDY
% ============================================================================
\begin{frame}{Case Study: PWM Actuation Integrity}
\textbf{Signals.}
\begin{itemize}
\item \keyword{Intent} (from shared record): duty, dead-time, update sequence, timestamp.
\item \keyword{Evidence:} edge timings / waveform-derived duty, glitch signatures, timing anomalies.
\end{itemize}

\vspace{0.6em}
\textbf{Example policies (illustrative).}
\begin{itemize}
\item Tracking: realized duty converges to intended duty within a deadline and stays within tolerance.
\item Glitch exclusion: no pulses below a minimum width.
\item Safety envelope: upon a safety condition, realized PWM enters a conservative envelope within a deadline.
\item Policies can be encoded in \STL or similar temporal formalisms; robust monitoring can output a margin.
\end{itemize}
\end{frame}

\begin{frame}{Tools and Interfaces We Will Explore}
\begin{itemize}
\item \keyword{Coherent interfaces:} Arm AMBA ACE/CHI to study coherent agent vs.\ coherent I/O attachment and access control for a shared region.

\end{itemize}
