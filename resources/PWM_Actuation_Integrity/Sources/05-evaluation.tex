\section{Evaluation}
\label{sec:evaluation}
\subsection{Experimental Setup and Metrics}
\begin{itemize}
    \item Operating points and conditions; what signals are logged; timing rates.
    \item Metrics: physical impact (tracking error, instability indicators, safety envelope violations), detection (TPR/FPR, detection latency), robustness (operating variation), and overhead.
\end{itemize}

\subsection{RQ1: Attack Effectiveness}
\begin{itemize}
    \item Demonstrate that actuation-path tampering and register corruption can cause closed-loop deviations (including transient single/few-cycle effects).
\end{itemize}

\subsection{RQ2: Detection/Attestation Performance}
\begin{itemize}
    \item TPR/FPR, latency (cycles), and robustness to benign variability.
\end{itemize}

\subsection{RQ3: Robustness and Ablations}
\begin{itemize}
    \item With/without conditioning on $(u_k,x_k,d_k)$; feature subset ablations; window size sensitivity.
    \item Stability under operating point changes and noise; (optionally) drift proxies.
\end{itemize}

\subsection{RQ4: Overheads and Deployability}
\begin{itemize}
    \item Runtime overhead and bandwidth; required measurement taps/sensors; verifier placement considerations.
\end{itemize}
